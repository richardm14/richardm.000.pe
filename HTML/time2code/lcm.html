# Least common multiple program 

# ------------------------- 

# Subprograms 

# ------------------------- 

def lcm(number1, number2): 

    if number1 > number2: 

        counter = number2 

    else: 

        counter = number1 

    while True: 

        if counter % number1 == 0 and counter % number2 == 0: 

            return counter 

        counter = counter + 1 

# ------------------------- 

# Main program 

# ------------------------- 

number1 = int(input("Enter the first number: ")) 

number2 = int(input("Enter the second number: ")) 

result = lcm(number1, number2) 

print("The LCM of", number1, "and", number2, "is", result)

# Compound Interest 

# ------------------------- 

# Subprograms 

# ------------------------- 



def compound(balance, interest_rate, target_balance): 

    year = 1 

    interest_rate = interest_rate / 100 

    while balance < target_balance: 

        interest = balance * interest_rate 

        balance = int(balance + interest) 

        print("Year", year, ": Balance £", balance) 

        year = year + 1
